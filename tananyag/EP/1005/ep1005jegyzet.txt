	indexelés

	lista[index] - 'index'-edik eleme a listának (0-tól számolunk)
	lista[index:ig] - 'index'-edik elemtõl 'ig'-1-ig
	lista[-1] - utolsó elem
	lista[5:-1] - 5-tõl utolsó elemig
	lista[:asd] - 0-tól asd-ig - asszem
	lista[asd:] - már nem emlékszem mi volt
